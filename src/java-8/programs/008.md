
# Write a java stream program to find 3rd largest number from given array

```java
import java.util.*;

public class Main {
    public static void main(String[] args) {
      
      // Optional<Integer> number=List.of(1,2,3,4,5).stream().sorted().skip(2).findFirst();
      // number.ifPresentOrElse(i -> System.out.println(i), () -> System.out.println("No value found"));
      
      nthLargest(List.of(1,1,2,2,3,4,5), 3);
      System.out.println("Hello, World!");
  }
  
  static void nthLargest(List<Integer> list, int position){
    
      Optional<Integer> number=list.stream().distinct().sorted().skip(position-1).findFirst();
      number.ifPresentOrElse(i -> System.out.println(i), () -> System.out.println("No value found"));
  }
}
```
