# Find the 3rd highest salary from the employees table without using LIMIT.

```sql

-- Find the 3rd highest salary from the employees table without using LIMIT.


CREATE TABLE employees (
  id INT PRIMARY KEY,
  name VARCHAR(50),
  dept VARCHAR(50),
  salary INT
);

INSERT INTO employees VALUES
(1, 'Clark', 'Sales', 5000),
(2, 'Dave', 'HR', 7000),
(3, 'Ava', 'IT', 9000),
(4, 'Rajan', 'Sales', 6000),
(5, 'Champa', 'IT', 8500),
(6, 'Jaggu', 'Finance', 7500);

select * from employees;


SELECT e.*
FROM employees e
WHERE 2 = (
   SELECT COUNT(DISTINCT e2.salary)
   FROM employees e2
   WHERE e2.salary > e.salary
);


```
