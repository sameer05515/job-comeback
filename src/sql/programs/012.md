
# Retrieve the longest order processing time from orders table (difference between orderDate and deliveryDate).

```sql

-- Retrieve the longest order processing time from orders table 
-- (difference between orderDate and deliveryDate).

-- create table
CREATE TABLE ORDERS (
  orderId INTEGER PRIMARY KEY,
  customerName TEXT NOT NULL,
  orderDate DATE NOT NULL,
  deliveryDate DATE
);

INSERT INTO ORDERS VALUES (1, 'Clark',   '2025-01-05', '2025-01-08');
INSERT INTO ORDERS VALUES (2, 'Rajan',   '2025-02-05', '2025-02-15');
INSERT INTO ORDERS VALUES (3, 'Anjum',   '2025-01-05', '2025-01-06');
INSERT INTO ORDERS VALUES (4, 'Champa',   '2025-01-05', '2025-02-08');



SELECT orderId, customerName, orderDate, deliveryDate,
       DATEDIFF(deliveryDate, orderDate) AS processing_days
FROM ORDERS
ORDER BY processing_days DESC
LIMIT 1;

```
