# Find duplicate email IDs in the users table.

```sql


-- Find duplicate email IDs in the users table.

-- create
CREATE TABLE USERS (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  email TEXT NOT NULL,
  phone TEXT
);



-- insert
INSERT INTO USERS VALUES (0001, 'Clark', 'clark@gmail.com',8010645625);
INSERT INTO USERS VALUES (0002, 'Dave', 'dave@gmail.com',8010645626);
INSERT INTO USERS VALUES (0003, 'Ava', 'clark@gmail.com',8010645627);
INSERT INTO USERS VALUES (0004, 'Champa', 'champa@gmail.com',8010645628);



-- fetch 
SELECT * FROM USERS ;

select email, count(*) duplicate_count from USERS group by email having duplicate_count>1;

--  If you also want full user details

select *
from USERS
where email in (
    select email 
    from USERS 
    group by email 
    having count(*) > 1
);


```
