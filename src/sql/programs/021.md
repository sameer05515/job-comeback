# Find pairs of users who logged in on the same day from the logins table.

```sql

-- Find pairs of users who logged in on the same day from the logins table.


CREATE TABLE logins (
  userId INT,
  loginDate DATE
);

INSERT INTO logins VALUES
(1, '2025-01-01'),
(2, '2025-01-01'),
(3, '2025-01-01'),
(1, '2025-01-02'),
(2, '2025-01-03');

SELECT l1.userId AS user1, l2.userId AS user2, l1.loginDate
FROM logins l1
JOIN logins l2 
  ON l1.loginDate = l2.loginDate 
 AND l1.userId < l2.userId
ORDER BY l1.loginDate, l1.userId, l2.userId;


```

```
+-------+-------+------------+

|     1 |     3 | 2025-01-01 |
|     2 |     3 | 2025-01-01 |
+-------+-------+------------+

```
