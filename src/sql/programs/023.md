# Find all orders that were delivered late (deliveryDate > promisedDate).

```sql

-- Find all orders that were delivered late (deliveryDate > promisedDate).

CREATE TABLE delivery_orders (
  id INT PRIMARY KEY,
  customer VARCHAR(50),
  orderDate DATE,
  promisedDate DATE,
  deliveryDate DATE
);

INSERT INTO delivery_orders VALUES
(1, 'Clark', '2025-01-01', '2025-01-05', '2025-01-06'),
(2, 'Dave', '2025-01-02', '2025-01-08', '2025-01-07'),
(3, 'Ava', '2025-01-03', '2025-01-10', '2025-01-15');



select * from delivery_orders where deliveryDate>promisedDate;

SELECT id, customer, orderDate, promisedDate, deliveryDate
FROM delivery_orders
WHERE deliveryDate > promisedDate;


-- a version that shows "On Time" vs "Late" for every order instead of filtering only lates

SELECT 
  id, 
  customer, 
  orderDate, 
  promisedDate, 
  deliveryDate,
  CASE 
    WHEN deliveryDate > promisedDate THEN 'Late'
    ELSE 'On Time'
  END AS delivery_status
FROM delivery_orders;

```
