# Find products that were sold but never restocked from products and restocks tables.

```sql

-- Find products that were sold but never restocked from products and restocks tables.


CREATE TABLE products (
  productId INT PRIMARY KEY,
  name VARCHAR(50)
);

CREATE TABLE sales (
  saleId INT PRIMARY KEY,
  productId INT,
  qty INT
);

CREATE TABLE restocks (
  restockId INT PRIMARY KEY,
  productId INT,
  qty INT
);

INSERT INTO products VALUES
(1, 'Laptop'), (2, 'Phone'), (3, 'Tablet');

INSERT INTO sales VALUES
(1, 1, 5), (2, 2, 10), (3,3,5);

INSERT INTO restocks VALUES
(1, 1, 5);  -- Phone, Tablet never restocked


-- query for products that were sold but never restocked ðŸ‘‡

SELECT DISTINCT p.productId, p.name
FROM products p
JOIN sales s ON p.productId = s.productId
LEFT JOIN restocks r ON p.productId = r.productId
WHERE r.productId IS NULL;


```
