-- customers who placed more than 5 orders in a single month ðŸ‘‡

SELECT customer, 
       YEAR(orderDate) AS order_year,
       MONTH(orderDate) AS order_month,
       COUNT(*) AS order_count
FROM orders
GROUP BY customer, YEAR(orderDate), MONTH(orderDate)
HAVING COUNT(*) > 5;
