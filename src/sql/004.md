# find all project names where employee name is 'Prem'

```sql


-- create
CREATE TABLE EMPLOYEE (
  empId INTEGER PRIMARY KEY,
  name TEXT NOT NULL,
  dept TEXT NOT NULL,
  salary INTEGER
);

CREATE TABLE PROJECT (
  id INTEGER PRIMARY KEY,
  name TEXT NOT NULL
);

CREATE TABLE EMPPROJECT (
  id INTEGER PRIMARY KEY,
  empId INTEGER,
  projId INTEGER
);

-- find all project names where employee name is 'Prem'
SELECT p.name from PROJECT p where p.id in 
(SELECT ep.projId from EMPLOYEE e join EMPPROJECT ep on e.empId=ep.empId where e.name='Prem');
```
